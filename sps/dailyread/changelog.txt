CREATE TABLE `dailyread` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT,
  `sid` tinyint(4) DEFAULT '0',
  `stu` varchar(18) DEFAULT NULL,
  `stuname` varchar(48) DEFAULT NULL,
  `tea` varchar(18) DEFAULT NULL,
  `teaname` varchar(48) DEFAULT NULL,
  `cls` varchar(18) DEFAULT NULL,
  `clsname` varchar(48) DEFAULT NULL,
  `sub` varchar(18) DEFAULT NULL,
  `subname` varchar(48) DEFAULT NULL,
  `year` varchar(16) DEFAULT NULL,
  `dt` date DEFAULT NULL,
  `tm` time DEFAULT NULL,
  `ms` integer DEFAULT '0',
  `sta` tinyint(4) DEFAULT '0',
  `des` varchar(128) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `sid` (`sid`),
  KEY `year` (`year`),
  KEY `stu` (`stu`),
  KEY `cls` (`cls`),
  KEY `sub` (`sub`),
  KEY `dt` (`dt`)
);
